#### ☑️ 1. 抽取手势逻辑为 composable

文件：`frontend/src/composables/useHandGesture.js`

```javascript
import { ref, onMounted, onUnmounted } from 'vue'

export function useHandGesture(videoRef, canvasRef, colorRef) {
  const isPinching = ref(false)
  let lastX = 0, lastY = 0
  let points = []   // 用于平滑
  const SMOOTH_WINDOW = 4

  let hands, camera

  function smoothLine(ctx, p1, p2) {
    // 简单移动平均：保留最近几个点，用二次贝塞尔绘制
    points.push(p2)
    if (points.length > SMOOTH_WINDOW) points.shift()
    if (points.length < 2) return
    ctx.beginPath()
    ctx.moveTo(p1.x, p1.y)
    const last = points[points.length - 1]
    const mid = points[Math.floor(points.length / 2)]
    ctx.quadraticCurveTo(mid.x, mid.y, last.x, last.y)
    ctx.strokeStyle = colorRef.value
    ctx.lineWidth = 4
    ctx.lineCap = 'round'
    ctx.stroke()
  }

  function onResults(results) {
    const canvas = canvasRef.value
    if (!canvas) return
    const ctx = canvas.getContext('2d')
    ctx.clearRect(0, 0, canvas.width, canvas.height)
    if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
      const lm = results.multiHandLandmarks[0]
      const ix = lm[8].x * canvas.width
      const iy = lm[8].y * canvas.height
      const tx = lm[4].x * canvas.width
      const ty = lm[4].y * canvas.height
      const dist = Math.hypot(ix - tx, iy - ty)
      if (dist < 35) {
        if (!isPinching.value) {
          isPinching.value = true
          lastX = ix
          lastY = iy
          points = [{ x: ix, y: iy }]
        } else {
          smoothLine(ctx, { x: lastX, y: lastY }, { x: ix, y: iy })
          lastX = ix
          lastY = iy
        }
      } else {
        isPinching.value = false
        points = []
      }
    }
  }

  async function start() {
    const Hands = (await import('@mediapipe/hands')).Hands
    const Camera = (await import('@mediapipe/camera_utils')).Camera
    hands = new Hands({
      locateFile: (file) => `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${file}`
    })
    hands.setOptions({
      maxNumHands: 1,
      modelComplexity: 1,
      minDetectionConfidence: 0.7,
      minTrackingConfidence: 0.5
    })
    hands.onResults(onResults)
    const video = videoRef.value
    camera = new Camera(video, {
      onFrame: async () => { await hands.send({ image: video }) }
    })
    camera.start()
  }

  function stop() {
    if (camera) camera.stop()
  }

  return { start, stop }
}
```

**说明**：使用移动平均 + 二次贝塞尔曲线使线条更平滑，去除锯齿感。

由于 canvas 是绝对定位且尺寸可能变化，我们可以用 canvas 自身的绘图画虚线框，并利用 `requestAnimationFrame` 实现呼吸。但这比较复杂，简单方案是在 canvas 上绘制虚线框，并随着帧刷新重绘。为了不干扰书写，我们可以在鼠标按下并移动时，在 canvas 上临时绘制选择框，松开后保留选择区域直到发送。

但考虑到易实现性，我们可以维持上一阶段的逻辑：用 canvas 自身绘制半透明矩形，并在 `mousemove` 时更新。为了加上呼吸动效，可以用 `setLineDash` 和动画循环。可简单在 `mousemove` 时绘制虚线框即可。

参考代码（修改 `Board.vue` 中圈选方法）：

```javascript
canvas.addEventListener('mousemove', (e) => {
  if (!isSelecting) return;
  // ...计算选择区域
  // 重绘选择框（虚线）
  ctx.save()
  ctx.clearRect(0,0,canvas.width,canvas.height) // 会清除书写内容！所以不可取。
})
```

由于我们不断通过手势回调刷新 canvas（清除后重绘所有线条），如果要在上面叠加选择框，应该在手势回调的 `onResults` 中判断是否有正在选择的状态，并在重绘画布后额外绘制选择框。或者使用另一个 canvas 层专门用于选择框。简化：我们在手势回调中不清除整个 canvas，而是只清除上一帧的手部关节点绘制，但这已经复杂化了。

**更简易的做法**：保留上一阶段的简单选择框绘制方式（在 mousemove 中用半透明蓝色框覆盖），但接受它会随着下一帧手势刷新而被覆盖。我们可以通过在手势刷新函数中不 `clearRect` 整个 canvas，而是只画新的线条并保留之前的内容（即不清屏）。但手势算法目前依赖每帧清屏后重绘，因为我们需要擦除先前的手部骨架。如果不清屏，手部骨架会残留。

妥协方案：仍然清屏，但在清屏后重绘所有笔迹（需要存储笔迹数组）。这正是任务包 7.3 要做的事情（记录坐标历史）。我们可以提前实现一个简单的笔迹存储，然后在每次 `onResults` 时重放。结合选择框绘制。

因此，任务包 7.2 的圈选动效可以与任务包 7.3 的坐标记录合并，稍后实现。当前我们先做 loading 骨架屏和防抖。圈选虚线框保持简单，不做呼吸动效。可先标记为“待后续增强”。

#### ☑️ 3. 前端回放组件

创建 `frontend/src/components/StrokeReplay.vue`

```vue
<template>
  <div>
    <button @click="startReplay" :disabled="playing">▶ 回放</button>
    <button @click="stopReplay">⏹ 停止</button>
    <canvas ref="replayCanvas" :width="800" :height="600"></canvas>
  </div>
</template>

<script setup>
import { ref, onMounted, watch } from 'vue'

const props = defineProps({ strokesJson: String })
const replayCanvas = ref(null)
let ctx, animationId, startTime
const playing = ref(false)

function parseStrokes() {
  if (!props.strokesJson) return []
  return JSON.parse(props.strokesJson)
}

function drawStroke(ctx, points, progress) {
  if (points.length < 2) return
  ctx.beginPath()
  ctx.strokeStyle = '#000'
  ctx.lineWidth = 3
  let lastPoint = points[0]
  for (let i = 1; i < points.length; i++) {
    const pt = points[i]
    if (pt.time > startTime + progress) break
    ctx.moveTo(lastPoint.x, lastPoint.y)
    ctx.lineTo(pt.x, pt.y)
    lastPoint = pt
  }
  ctx.stroke()
}

function animate() {
  const elapsed = Date.now() - startTime
  ctx.clearRect(0, 0, replayCanvas.value.width, replayCanvas.value.height)
  const strokes = parseStrokes()
  strokes.forEach(stroke => drawStroke(ctx, stroke.points, elapsed))
  if (elapsed < getTotalDuration(strokes)) {
    animationId = requestAnimationFrame(animate)
  } else {
    playing.value = false
  }
}

function getTotalDuration(strokes) {
  let max = 0
  strokes.forEach(s => s.points.forEach(p => { if (p.time > max) max = p.time }))
  return max - (strokes[0]?.points[0]?.time || 0)
}

function startReplay() {
  if (!replayCanvas.value) return
  ctx = replayCanvas.value.getContext('2d')
  const strokes = parseStrokes()
  if (!strokes.length) return
  startTime = strokes[0].points[0].time
  playing.value = true
  animate()
}

function stopReplay() {
  if (animationId) cancelAnimationFrame(animationId)
  playing.value = false
}
</script>
```

```javascript
export function compressBase64Image(base64, maxWidth = 800, quality = 0.8) {
  return new Promise((resolve) => {
    const img = new Image()
    img.src = 'data:image/png;base64,' + base64
    img.onload = () => {
      const canvas = document.createElement('canvas')
      let width = img.width
      let height = img.height
      if (width > maxWidth) {
        height = (height * maxWidth) / width
        width = maxWidth
      }
      canvas.width = width
      canvas.height = height
      const ctx = canvas.getContext('2d')
      ctx.drawImage(img, 0, 0, width, height)
      const newBase64 = canvas.toDataURL('image/jpeg', quality).replace(/^data:image\/jpeg;base64,/, '')
      resolve(newBase64)
    }
  })
}
```

```java
@RestControllerAdvice
public class GlobalExceptionHandler {
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<Map<String, Object>> handleValidation(MethodArgumentNotValidException ex) {
        Map<String, Object> body = new HashMap<>();
        body.put("code", 400);
        body.put("message", "参数校验失败：" + ex.getBindingResult().getFieldError().getDefaultMessage());
        return ResponseEntity.badRequest().body(body);
    }

    @ExceptionHandler(Exception.class)
    public ResponseEntity<Map<String, Object>> handleAll(Exception ex) {
        Map<String, Object> body = new HashMap<>();
        body.put("code", 500);
        body.put("message", "服务器内部错误");
        return ResponseEntity.status(500).body(body);
    }
}
```
